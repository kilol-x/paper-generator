<template>
<div class="preview-container">
<!-- 顶部操作栏 -->
<div class="preview-toolbar">
<el-button @click="goBack" icon="ArrowLeft">返回编辑</el-button>
<el-button type="primary" @click="exportDocx" icon="Download">
导出 Word
</el-button>
<el-button type="success" @click="exportPdf" icon="Download">
导出 PDF
</el-button>
<el-button @click="toggleFullscreen" icon="FullScreen">
全屏预览
</el-button>
<span class="page-info">第 {{ currentPage }} / {{ totalPages }} 页</span>
</div>

<!-- 分页控制器 -->
<div class="page-controls">
<el-button @click="prevPage" :disabled="currentPage <= 1">上一页</el-button>
<el-pagination
small
:page-size="1"
:total="totalPages"
:current-page="currentPage"
@current-change="goToPage"
layout="prev, pager, next"
/>
<el-button @click="nextPage" :disabled="currentPage >= totalPages">下一页</el-button>
</div>

<!-- 论文预览 - 模拟A4纸 -->
<div class="paper-wrapper" ref="paperWrapper">
<div class="paper-page" v-for="(page, index) in pages" 
v-show="index + 1 === currentPage"
:key="index"
id="paper-content">
<div v-html="page"></div>
</div>
</div>
</div>
</template>

<script setup>
import { ref, onMounted, computed } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { ElMessage } from 'element-plus'
import axios from 'axios'
import html2canvas from 'html2canvas'
import jsPDF from 'jspdf'

const route = useRoute()
const router = useRouter()
const paperId = ref(route.params.id)

// 状态
const paperData = ref(null)
const pages = ref([])
const currentPage = ref(1)
const totalPages = ref(0)

// API 获取论文数据
const fetchPaperData = async () => {
	try {
		const res = await axios.get(`/api/student/papers/${paperId.value}/preview`)
		paperData.value = res.data
		// 按A4纸分页，每页内容拆分为独立HTML
		pages.value = splitIntoPages(res.data.fullHtml)
		totalPages.value = pages.value.length
	} catch (error) {
		ElMessage.error('加载论文数据失败')
	}
}

// 分页逻辑 - 按A4纸高度拆分内容
const splitIntoPages = (html) => {
	// 实际项目中需要根据内容高度动态分页
	// 这里简化实现，用占位符代替真实分页逻辑
	const pageSize = 1000 // 字符数，实际应基于渲染高度
	const result = []
	const content = html
	for (let i = 0; i < content.length; i += pageSize) {
		result.push(content.substring(i, i + pageSize))
	}
	// 如果为空，返回单页占位
	if (result.length === 0) result.push('<div>暂无内容</div>')
	return result
}

// 翻页
const prevPage = () => {
	if (currentPage.value > 1) currentPage.value--
}
const nextPage = () => {
	if (currentPage.value < totalPages.value) currentPage.value++
}
const goToPage = (page) => {
	currentPage.value = page
}
const goBack = () => {
	router.push(`/student/paper/edit/${paperId.value}`)
}

// 全屏
const toggleFullscreen = () => {
	const el = document.querySelector('.paper-wrapper')
	if (!document.fullscreenElement) {
		el.requestFullscreen()
	} else {
		document.exitFullscreen()
	}
}

// ============================================
// 🚀 docx 导出（核心）
// ============================================
const exportDocx = async () => {
	try {
		ElMessage.info('正在生成 Word 文档，请稍候...')
		
		const response = await axios({
			url: `/api/student/papers/${paperId.value}/export/docx`,
			method: 'GET',
			responseType: 'blob' // 重要：接收二进制文件
		})
		
		// 创建下载链接
		const url = window.URL.createObjectURL(new Blob([response.data]))
		const link = document.createElement('a')
		link.href = url
		link.download = `${paperData.value?.title || '论文'}.docx`
		document.body.appendChild(link)
		link.click()
		document.body.removeChild(link)
		window.URL.revokeObjectURL(url)
		
		ElMessage.success('Word 文档导出成功')
	} catch (error) {
		ElMessage.error('导出 Word 失败，请检查网络或重试')
		console.error(error)
	}
}

// ============================================
// 🚀 PDF 导出（核心）
// ============================================
const exportPdf = async () => {
	try {
		ElMessage.info('正在生成 PDF 文档，请稍候...')
		
		// 方案一：直接请求后端生成 PDF
		const response = await axios({
			url: `/api/student/papers/${paperId.value}/export/pdf`,
			method: 'GET',
			responseType: 'blob'
		})
		
		const url = window.URL.createObjectURL(new Blob([response.data]))
		const link = document.createElement('a')
		link.href = url
		link.download = `${paperData.value?.title || '论文'}.pdf`
		document.body.appendChild(link)
		link.click()
		document.body.removeChild(link)
		window.URL.revokeObjectURL(url)
		
		ElMessage.success('PDF 文档导出成功')
	} catch (error) {
		ElMessage.error('导出 PDF 失败，请检查网络或重试')
		console.error(error)
	}
}

onMounted(() => {
	fetchPaperData()
})
</script>
