\section{引言}
\subsection{编写目的}
	本文档为"在线论文文档自动生成工具"软件需求规格说明书，用于完整定义系统面向用户的功能需求与业务规则，明确项目边界与使用约束。
    本文档交付对象为项目经理、业务方、教师与学生用户，作为系统功能验收与使用培训的核心依据。
    本文档的读者对象包括：
    \begin{itemize}
	    \item 项目管理人员：了解系统整体功能范围与交付边界，用于项目规划与进度管控；
	    \item 业务方/甲方代表：确认系统功能是否满足实际业务需求，作为验收依据；
	    \item 测试人员：依据功能需求条目编写测试用例，验证系统功能完整性与正确性；
	    \item 最终用户（教师、学生、管理员）：了解系统提供的功能与操作方式，作为使用参考。
    \end{itemize}

\subsection{项目背景}
    本系统为高校大二软件工程实训项目，开发单位为在校实训开发小组，服务对象为高校教师与在校学生。
    在当前高校教学实践中，学生撰写课程论文与毕业论文时普遍面临以下问题：
    \begin{enumerate}
	    \item 论文格式不统一，不同学院、不同课程的论文模板差异较大，学生排版耗时且容易出错；
	    \item Word 排版工具门槛较高，部分学生不熟悉样式、分节符、页眉页脚等高级功能，导致格式反复修改；
	    \item 教师批阅纸质或电子版论文时，批注留痕不系统，修改意见分散，难以追溯历史版本；
	    \item 论文模板由各学院自行管理，缺乏统一的模板配置与分发机制，更新不及时。
    \end{enumerate}
    针对上述问题，本系统旨在提供一套基于 Web 的在线论文编辑与标准化文档生成工具，实现从模板配置、在线写作、格式预览、文档导出到教师批阅的完整业务闭环。
	
\subsection{定义}
	\begin{itemize}
        \item \textbf{富文本编辑器}：网页端所见即所得的文本编辑组件，支持字体、层级、表格、图片嵌入等排版操作；
        \item \textbf{docx}：Microsoft Word 文档标准格式；
	    \item \textbf{PDF}：便携式文档格式，用于跨平台保持文档排版一致性；
		\item \textbf{论文模板}：预设的论文格式规范，包含章节结构、字体字号、行距页边距等排版参数；
		\item \textbf{参考文献}：论文中引用他人研究成果的文献条目，系统按标准格式自动排版；
		\item \textbf{批注}：教师在批阅论文时添加的修改意见或建议，关联到文档具体位置；
		\item \textbf{草稿保存}：系统在编辑过程中自动保存当前内容，防止意外丢失；
		\item \textbf{三线表}：学术论文常用的标准表格格式，由顶线、栏目线和底线三条横线构成。
	\end{itemize}

\section{任务概述}
\subsection{目标}
	\begin{enumerate}
		\item \textbf{统一模板管理}：各学院可配置专属的论文格式模板，覆盖毕业论文、课程论文、项目设计等多种类型，确保全院学生使用一致的排版规范；
		\item \textbf{在线便捷写作}：学生通过网页即可进行论文的在线编辑，支持文字排版、图片插入、表格编辑、参考文献管理等常用功能，系统自动保存写作进度；
		\item \textbf{标准化文档导出}：编辑完成后，学生可一键导出符合学校格式规范的Word文档与PDF文件，无需手动调整排版。
		\item \textbf{课程论文批阅}：教师可在线查看学生提交的课程论文，添加批注与评语，评定成绩，学生可查看批阅意见并修改后重新提交。
	\end{enumerate}

\subsection{运行环境}
    系统为Web端应用，用户通过主流浏览器访问即可使用，无需安装任何客户端软件。
    支持的浏览器包括Chrome 90及以上版本、Edge 90及以上版本、Firefox 88及以上版本。
    建议用户在分辨率不低于1366×768的屏幕上使用，以获得最佳编辑体验。
	
\subsection{条件与限制}
	\begin{enumerate}
		\item 单篇论文正文字数上限为5万字，超出此限制需拆分处理；
		\item 单篇论文最多插入50张图片，单张图片大小不超过5MB；
		\item 参考文献条目上限为200条；
		\item 教师单条批注字数不超过500字符；
		\item 系统暂不支持复杂数学公式的在线编辑与排版；
		\item 系统暂不支持多人同时编辑同一篇论文；
		\item 导出的Word和PDF文档排版严格遵循预设模板规范，用户无法在导出后通过系统修改格式参数。
	\end{enumerate}
    本系统\textbf{不}具备以下功能，超出范围的需求不在本次开发目标内：
    \begin{enumerate}
	\item 不提供论文查重功能，系统不检测论文内容的原创性；
	\item 不提供 AI 写作辅助功能，论文内容完全由学生自主撰写；
	\item 不提供学术不端检测功能，抄袭判定由教师或学校其他系统负责；
	\item 不对接学校教务系统，课程信息与学生名单不自动同步；
	\item 不提供论文内容的版本差异对比功能（如 Word 的修订模式）。
	\end{enumerate}

\subsection{动态数据}

\subsection{数据采集}
    \begin{itemize}
       \item \textbf{用户信息}：学生通过注册页面自行填写提交；教师和管理员信息由系统管理员在后台录入。
       \item \textbf{学院信息}：由系统管理员初始化配置，可根据学校院系调整进行增删改。
       \item \textbf{模板数据}：由系统管理员根据各学院的论文格式规范进行配置，包括章节结构和排版参数的设置。
       \item \textbf{论文内容}：由学生在在线编辑器中逐章填写、上传图片、插入表格、录入参考文献，系统自动采集并保存。
       \item \textbf{批阅数据}：教师在批阅界面中添加批注、填写评语和评分时产生，由系统自动记录。
       \item \textbf{操作日志}：系统后台自动记录用户的关键操作行为，包括登录、论文创建与修改、提交、导出、批阅等，用于安全审计和问题追溯。
    \end{itemize}



\section{功能需求}	
\subsection{功能划分}
	系统划分为以下六大核心功能模块：
	\begin{enumerate}
		\item 用户认证与权限管理模块；
		\item 后台基础数据管理模块；
		\item 论文模板管理模块；
		\item 学生在线写作模块；
		\item 文档预览与导出模块；
		\item 课程论文批阅模块。
	\end{enumerate}
	
\subsection{功能描述}	
	\subsubsection{用户认证与权限管理}
    该模块负责用户的注册、登录与权限隔离，是系统安全运行的基础保障。

	    \reqitem{FR001}{系统支持学生通过用户名和密码自行注册账号，需填写真实姓名、学号、邮箱、手机号及所属学院，系统校验用户名与邮箱唯一性。}
	
         \reqitem{FR002}{系统支持所有用户通过用户名密码登录，登录成功签发 JWT Token，前端存储 Token 并携带于后续请求中。}
	
        \reqitem{FR003}{系统基于 RBAC 模型实现权限拦截，对三类角色进行严格访问隔离，禁止越权操作。}
	
        \reqitem{FR004}{JWT Token 有效期为 2 小时，过期需重新登录，持续操作时自动续期。}
	
	\subsubsection{后台基础数据管理}
    该模块由管理员操作，维护学院基础信息，为模板分类提供数据支撑。

        \reqitem{FR005}{管理员可查看所有学院列表，列表以表格形式展示学院名称、描述、创建时间等信息，支持按学院名称关键字进行模糊搜索，搜索结果实时更新。}
    
        \reqitem{FR006}{管理员可新增学院信息，填写学院名称（必填，不可与已有学院重复）和描述信息（选填，简要说明该学院）。新增成功后，学院立即出现在列表中。}
    
        \reqitem{FR007}{管理员可编辑已有学院的名称和描述信息。编辑后保存即可生效，已关联该学院的模板不受影响。}
    
        \reqitem{FR008}{管理员可删除学院，删除前系统弹出二次确认对话框，明确提示"删除后无法恢复"。确认删除后，学院从列表中移除。}

    \subsubsection{论文模板管理}
	该模块是管理员端的核心功能，负责论文模板的全生命周期管理。模板定义了论文的格式规范，学生创建论文时自动套用所选模板的格式。
    
         \reqitem{FR009}{管理员可查看所有模板列表，列表以表格形式展示模板名称、类型、所属学院、启用状态、版本号、更新时间等信息。支持按学院、模板类型（毕业论文/课程论文/项目论文）、启用状态（已启用/已停用）进行组合筛选，搜索结果实时更新。}
    
        \reqitem{FR010}{管理员可新增论文模板，填写基本信息并配置模板内容。新增时需完成以下配置：}
    \begin{itemize}
        \item 模板名称：唯一，便于学生识别；
        \item 模板类型：选择毕业论文、课程论文或项目论文；
        \item 所属学院：选择模板适用的学院范围；
        \item 章节结构：定义论文的章节层级和标题，如"摘要"、"第一章"、"第二章"等；
        \item 排版参数：设置字体、字号、行距、页边距、页眉页脚样式等。
    \end{itemize}
        \reqitem{FR011}{管理员可编辑已有模板的基本信息和章节结构，但已启用的模板不允许修改排版参数，以保证学生创建论文时格式一致。编辑后保存即可生效。}
    
        \reqitem{FR012}{管理员可启用或停用模板，启用后学生可选择该模板创建论文，停用后新建论文无法选择该模板，但已创建的论文不受影响。}
    
        \reqitem{FR013}{管理员可删除未启用的模板，删除前系统弹出二次确认对话框，明确提示"删除后无法恢复"。确认删除后，模板从列表中移除。}
    
        \reqitem{FR014}{系统为每个模板维护版本号，每次修改章节结构或排版参数时自动生成新版本，历史版本保留以便回溯。}
    
        \reqitem{FR015}{管理员可导入/导出模板配置文件（JSON格式），便于在不同环境间迁移模板数据。导入时系统校验文件格式和字段完整性，导出时生成包含所有模板信息的JSON文件。}

	\subsubsection{学生在线写作}
	该模块是学生端的核心功能，学生在此完成论文的全部编辑工作。

        \reqitem{FR016}{学生可按学院与论文类型筛选可用模板，从已启用的模板中选择一个，点击创建论文项目。创建时系统自动根据所选模板生成预设的章节结构，学生即可开始编辑。}
    
        \reqitem{FR017}{编辑器支持以下文字排版操作：标题层级设置（一级/二级/三级标题）、文字加粗、文字斜体、下划线、左对齐/居中/右对齐/两端对齐、段落缩进、有序列表/无序列表。编辑器为所见即所得模式，学生可直接在页面上编辑内容。}
    
        \reqitem{FR018}{学生可从本地上传图片（支持 JPG、PNG、GIF、SVG 格式），上传后图片嵌入文档指定位置。单张图片大小不超过 5MB，单篇论文图片总数不超过 50 张。图片上传失败时，系统提示具体原因（如格式不支持、文件过大等）。}
    
        \reqitem{FR019}{学生可在文档中插入自定义表格，支持设置行列数、合并拆分单元格、调整列宽行高。系统提供学术三线表模板，学生可一键将普通表格转换为三线表格式，满足学术论文的表格规范要求。}
    
        \reqitem{FR020}{学生可通过独立的参考文献管理模块逐条录入文献信息，每条文献包含以下字段：文献类型（期刊论文/专著/学位论文/会议论文/网页等）、作者、标题、刊物名称、出版年份、卷号、期号、页码。系统按照模板定义的参考文献格式（默认 [N] 编号格式）自动统一排版参考文献列表。}
    
        \reqitem{FR021}{学生可在正文中插入引用标注，系统自动生成带方括号的序号（如 [1]、[2]），并与参考文献条目关联。引用标注的编号随参考文献列表的顺序自动更新。}
    
        \reqitem{FR022}{系统每 30 秒自动保存一次论文草稿，同时学生可随时点击"保存"按钮手动保存。自动保存在后台静默执行，不影响学生的编辑操作。学生可多次退出并重新进入系统，上次保存的内容会自动加载，支持续写修改。}

	\subsubsection{全文预览与文档导出}
	该模块为学生提供论文的最终效果查看与文件下载功能。
    
        \reqitem{FR023}{学生可一键生成整篇论文的网页预览页面。预览页面完整展示论文的封面、各级章节、图片、表格、参考文献的排版效果，模拟 A4 纸张样式，支持分页浏览。预览页面为只读模式，不可编辑。}
        \reqitem{FR024}{系统按模板既定格式，一键导出标准 docx 格式 Word 文档。导出的 Word 文档完整保留模板定义的排版格式（字体、字号、行距、页边距、页眉页脚等），图片与表格正确嵌入，参考文献按模板格式统一排版。同时支持导出 PDF 格式文件，PDF 排版与 docx 保持一致。}
        \reqitem{FR025}{导出功能支持选择导出范围：可选择导出整篇论文（包含封面、摘要、正文、参考文献等全部章节），或仅导出正文部分（从第一章开始到最后一章结束）。}

    \subsubsection{课程论文批阅}
	该模块实现学生提交课程论文后，教师在线批阅、评分、退回的完整业务流程。

        \reqitem{FR026}{学生完成课程论文编辑后，可将论文提交给对应授课教师。提交后，该论文进入锁定状态，学生不可继续编辑。提交操作不可撤销，学生需确认后再执行。}
        \reqitem{FR027}{教师可在线查看学生提交的论文预览内容，完整浏览论文排版效果。教师可在论文的任意段落或行位置添加批注，批注内容为修改建议或问题说明。每条批注不超过 500 个字符。}
        \reqitem{FR028}{教师可填写整体评语，评定论文分数（0-100 分）与等级（优秀/良好/中等/及格/不及格）。教师执行"通过"操作表示论文合格，执行"退回"操作时必须填写退回原因与修改建议，退回后学生可继续修改。}
        \reqitem{FR029}{系统留存每一次提交、批阅、退回的版本记录。学生可查看教师添加的所有批注内容、评语与评分，根据修改意见调整论文后可重新提交复审。每次重新提交，教师可查看此前所有批阅历史，了解论文的修改过程。}


	
\section{性能需求}
\subsection{数据精确度}
系统需确保用户录入的数据与最终展示、导出结果完全一致。参考文献序号与正文引用标注须一一准确对应，不得出现错位或遗漏。导出文档的排版格式须与模板定义保持一致，页面布局、字体字号、行距段落等格式参数误差控制在合理范围内。

\subsection{时间特性}
系统关键操作的响应时间要求如下：页面初次加载不超过2秒；论文保存操作即时反馈；网页全文预览生成不超过3秒；Word文档导出不超过5秒，PDF文档导出不超过8秒（以万字论文为参考）。在正常使用条件下，系统应保持良好的交互流畅度，不出现明显的操作卡顿。

\subsection{适应性}
系统须兼容主流浏览器的最新两个主要版本，适配常见屏幕分辨率（包括1920×1080、1366×768等），确保不同设备上的用户均能获得一致的编辑与阅读体验。



\section{运行需求}
\subsection{用户界面}
系统界面设计遵循简洁清晰的原则，各类用户角色的主要界面布局如下：
\begin{itemize}
\item \textbf{登录/注册页面}：居中卡片式布局，输入项清晰标注，提供友好的错误提示；
\item \textbf{学生工作台}：左侧导航菜单展示论文列表与各章节入口，右侧为内容编辑区；
\item \textbf{论文编辑器}：顶部工具栏提供常用排版功能，中部为编辑区域，底部显示保存状态与操作按钮；
\item \textbf{模板管理后台}：表格化展示模板列表，提供搜索、筛选与快捷操作入口；
\item \textbf{论文预览页}：模拟A4纸张样式，完整展示分页效果，提供导出按钮；
\item \textbf{批阅界面}：左侧展示论文内容，右侧为批注面板，底部为评语与评分表单；
\end{itemize}

\subsection{硬件接口}
无需专用硬件外设接口，仅依赖服务器CPU、内存、磁盘资源完成代码编译与数据存储。

\subsection{软件接口}
	\begin{itemize}
		\item \textbf{数据库接口}：JDBC 连接 MySQL，HikariCP 连接池；
		\item \textbf{文件存储}：Java NIO 处理上传，存储至本地文件系统；
		\item \textbf{文档生成}：Apache POI 生成 docx，iText 生成 PDF；
		\item \textbf{前端交互}：Axios 封装 HTTP 请求，JSON 格式交互；
		\item \textbf{缓存}：可选 Redis 存储 Token 黑名单与热点数据。
	\end{itemize}
	
\subsection{外部接口}
	系统为独立部署应用，预留以下可选接口：
	\begin{itemize}
		\item \textbf{邮件服务接口}（SMTP）：用于发送注册验证、密码重置、提交通知等邮件；
		\item \textbf{统一身份认证接口}（OAuth 2.0/CAS）：可选对接学校统一身份认证系统；
		\item \textbf{云存储接口}（阿里云 OSS/腾讯云 COS）：可选用于存储论文附件图片。
	\end{itemize}

\subsection{故障处理}
	\begin{itemize}
	\item \textbf{网络请求失败}：弹窗提示错误原因，保留用户当前编辑内容，用户可关闭弹窗后重试；
	\item \textbf{文件上传失败}：弹窗提示具体原因（格式不支持、文件过大、网络超时等），保留用户编辑内容；
	\item \textbf{文档导出超时}：提示"导出任务超时，请稍后重试"；
	\item \textbf{网络中断}：前端提示"网络已断开"，恢复后用户可手动刷新页面重试；
	\item \textbf{服务器异常}：提示"服务器暂时不可用，请稍后重试"。
	\end{itemize}
	


\section{非功能性需求}
\subsection{安全性需求}
	\begin{itemize}
		\item \textbf{密码安全}：用户密码采用 BCrypt 强哈希加密（工作因子大于等于10），严禁明文存储；
		\item \textbf{传输安全}：生产环境启用 HTTPS，TLS 1.2 及以上加密传输；
		\item \textbf{注入防范}：数据库操作使用参数化查询或 ORM，禁止拼接 SQL；
		\item \textbf{XSS/CSRF 防范}：用户输入严格过滤转义，状态变更请求验证 CSRF Token；
		\item \textbf{越权防范}：完整垂直与水平越权防范，学生仅可编辑本人论文，教师仅可批阅本人课程论文；
		\item \textbf{会话安全}：JWT Token 设置 2 小时过期，使用安全密钥签名；
		\item \textbf{文件安全}：限制上传文件类型为图片格式，校验 MIME 类型，重命名存储；
		\item \textbf{审计日志}：记录登录、论文创建/修改/提交/导出、模板管理、批阅等核心操作日志，保留 180 天。
	\end{itemize}
	
\subsection{可靠性需求}
	\begin{itemize}
		\item 系统全年可用性不低于 99.5\%；
		\item 数据库每日凌晨自动全量备份，保留 30 天；
		\item 系统具备故障自动恢复能力，重启后数据不丢失；
		\item 自动保存机制确保最多丢失 30 秒内编辑内容；
		\item 关键业务操作（提交、批阅、导出）具备事务性，保证数据一致性。
	\end{itemize}
	
\subsection{可维护性需求}
	\begin{itemize}
		\item 代码遵循统一编码规范，具备良好注释覆盖率；
		\item 集成标准日志框架（SLF4J + Logback），日志按级别分类，按日期滚动归档；
		\item 系统提供健康检查接口（/actuator/health）；
		\item 配置支持外部化，开发/测试/生产环境隔离；
		\item 核心模块单元测试覆盖率不低于 70\%；
	\end{itemize}

\subsection{易用性需求}
	\begin{itemize}
		\item 界面遵循 Material Design 规范，操作反馈清晰明确；
		\item 支持 Ctrl+S 保存草稿、Ctrl+B 加粗等键盘快捷键；
		\item 适配主流屏幕分辨率（1920×1080、1366×768）。
	\end{itemize}
	
\subsection{可扩展性需求}
	\begin{itemize}
		\item 应用服务器与数据库支持集群部署，通过负载均衡实现水平扩展；
		\item 模板类型支持扩展（可增加调研报告、实验报告等）；
		\item 参考文献格式支持扩展（预留 GB/T 7714、APA、MLA、Chicago 等配置接口）；
		\item 文档导出引擎支持插件化扩展（后续可增加 LaTeX、HTML 等格式）。
	\end{itemize}
